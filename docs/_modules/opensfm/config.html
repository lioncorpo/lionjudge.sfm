[docs]def load_config(filepath):
    """Load config from a config.yaml filepath"""
    config = default_config()

    if os.path.isfile(filepath):
        with open(filepath) as fin:
            new_config = yaml.safe_load(fin)
        if new_config:
            for k, v in new_config.items():
                config[k] = v

    return config
